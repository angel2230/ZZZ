ALTER PROC [dbo].[USP_RZ_ACCOUNT_INSERT]
(
	@USER_ID 		NVARCHAR(32)
  , @SITE_CODE 		NVARCHAR(32)
  , @SITE_USER_ID 	NVARCHAR(32)
  , @PWD 			NVARCHAR(32)
)
AS
	SET NOCOUNT ON;

	DECLARE @GSN BIGINT;
    EXEC dbo.USP_RZ_GLOBAL_ID @GSN OUTPUT;
    
    declare @MD5FromPwd varchar(100)
	exec RZ_ACCOUNTDB.dbo.USP_RZ_CREATEMD5WARZ @SITE_CODE, @MD5FromPwd OUTPUT

	INSERT INTO dbo.RZ_GT_USER (GSN, USER_ID, PWD, SITE_USER_ID, SITE_CODE)
	VALUES (@GSN, @USER_ID, @MD5FromPwd, @USER_ID, @MD5FromPwd);

    SELECT @GSN AS ACCN_ID;
